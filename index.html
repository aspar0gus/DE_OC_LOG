        // Initialize audio players

        function initAudioPlayer(player) {

            // Play/Pause

            player.playBtn.addEventListener('click', (e) => {

                e.preventDefault();

                player.audio.play();

                player.playBtn.style.display = 'none';

                player.pauseBtn.style.display = 'block';

            });


            player.pauseBtn.addEventListener('click', (e) => {

                e.preventDefault();

                player.audio.pause();

                player.pauseBtn.style.display = 'none';

                player.playBtn.style.display = 'block';

            });


            // Progress bar

            player.audio.addEventListener('timeupdate', () => {

                const currentTime = player.audio.currentTime;

                const duration = player.audio.duration || 1;

                const progressPercent = (currentTime / duration) * 100;

                player.progressFilled.style.width = `${progressPercent}%`;

                

                // Update time display

                const currentMinutes = Math.floor(currentTime / 60);

                const currentSeconds = Math.floor(currentTime % 60).toString().padStart(2, '0');

                const durationMinutes = Math.floor(duration / 60);

                const durationSeconds = Math.floor(duration % 60).toString().padStart(2, '0');

                player.timeDisplay.textContent = `${currentMinutes}:${currentSeconds} / ${durationMinutes}:${durationSeconds}`;

            });


            // Click on progress bar to seek

            player.progress.addEventListener('click', (e) => {

                e.preventDefault();

                const rect = player.progress.getBoundingClientRect();

                const pos = (e.clientX - rect.left) / rect.width;

                player.audio.currentTime = pos * player.audio.duration;

            });


            // Volume control

            player.volumeControl.addEventListener('input', (e) => {

                e.preventDefault();

                player.audio.volume = e.target.value;

                if (player.audio.volume > 0) {

                    player.audio.muted = false;

                    player.muteBtn.innerHTML = '<i class="fas fa-volume-up"></i>';

                }

            });


            // Mute button

            player.muteBtn.addEventListener('click', (e) => {

                e.preventDefault();

                player.audio.muted = !player.audio.muted;

                if (player.audio.muted) {

                    player.muteBtn.innerHTML = '<i class="fas fa-volume-mute"></i>';

                } else {

                    player.muteBtn.innerHTML = '<i class="fas fa-volume-up"></i>';

                }

            });

                e.preventDefault();

                if (e.target.files.length) {

                    const file = e.target.files[0];

                    const url = URL.createObjectURL(file);

                    player.audio.src = url;

                    player.songName ? player.songName.textContent = file.name : null;

        // Initialize drag and drop

        function initDragDrop() {

            // Handle drag over

            function handleDragOver(e) {

                e.preventDefault();

                this.style.borderColor = '#EBAC00';

                this.style.borderColor = '#EBAC00';

                e.preventDefault();

                this.style.borderColor = '#EBAC00';

                this.style.backgroundColor = 'rgba(62, 132, 237, 0.2)';

                            handleMultipleUpload(input.files, box);

                        }

                    }

            });

        }


        // Handle image upload

        function handleImageUpload(file, box) {

            if (!file.type.match('image.*')) {

                alert('Please upload an image file');

                return;

            }

            

            const reader = new FileReader();

            reader.onload = (e) => {

                displayImage(e.target.result, box);

            };

            reader.readAsDataURL(file);

        }


        // Handle multiple files upload

        function handleMultipleUpload(files, box) {

            box.innerHTML = '';

            

            // Create a container for multiple items

            const container = document.createElement('div');

            container.style.display = 'grid';

            container.style.gridTemplateColumns = 'repeat(auto-fill, minmax(100px, 1fr))';

            container.style.gap = '10px';

            container.style.width = '100%';

            container.style.height = '100%';

            container.style.overflow = 'auto';

            container.style.padding = '10px';

            

            Array.from(files).forEach(file => {

                if (file.type.match('image.*')) {

                    const reader = new FileReader();

                    reader.onload = (e) => {

                        const img = document.createElement('img');

                        img.src = e.target.result;

                        img.style.maxWidth = '100%';

                        img.style.maxHeight = '100px';

                        img.style.objectFit = 'cover';

                        img.style.borderRadius = '5px';

                        container.appendChild(img);

                    };

                    reader.readAsDataURL(file);

                }

            });

            

            box.appendChild(container);

            

            // Add delete button

            const deleteBtn = document.createElement('button');

            deleteBtn.className = 'delete-media-btn';

            deleteBtn.innerHTML = '×';

                e.stopPropagation();

        }

        function deleteCharacter(index) {

            if (confirm(`Are you sure you want to delete "${characters[index].name}"? This cannot be undone.`)) {

                const deletedName = characters[index].name;

                characters.splice(index, 1);

                localStorage.setItem('oc-characters', JSON.stringify(characters));

                renderGallery();

                updateObjectFilter();

                updateStatistics();

                

                // If we're editing this character, clear the form

                if (currentEditIndex === index) {

                    clearForm();

                    currentEditIndex = -1;

                } else if (currentEditIndex > index) {

                    // Adjust current edit index if it's after the deleted character

                    currentEditIndex--;

                }

                

                alert(`Character "${deletedName}" has been deleted.`);

            }

        }

        }


        // Filter them 

        function filterCharacters() {

            const nameSearch = searchNameInput.value.toLowerCase();

            const genderFilter = filterGenderSelect.value;

            const objectFilter = filterObjectSelect.value;

            

            const filtered = characters.filter(character => {

                const nameMatch = character.name.toLowerCase().includes(nameSearch);

                const genderMatch = !genderFilter || character.gender?.toLowerCase() === genderFilter;

                const objectMatch = !objectFilter || character.object?.toLowerCase() === objectFilter.toLowerCase();

                

                return nameMatch && genderMatch && objectMatch;

            });

            

            renderGallery(filtered);

        }


        // Update stats thing

        function updateStatistics() {

            document.getElementById('total-characters').textContent = characters.length;

            

            const genderCounts = {

                male: 0,

                female: 0,

                'non-binary': 0,

                other: 0

            };

            

            const ageGroups = {

                '0-17': 0,

                '18-20': 0,

                '21-25': 0,

                '26-30': 0,

                '31-35': 0,

                '36-40': 0,

                '41-45': 0,

                '46-50': 0,

                '51+': 0

            };

            

            const objectCounts = {};

            let totalAge = 0;

            let ageCount = 0;

            

            characters.forEach(character => {

                // Count genders

                if (character.gender) {

                    const gender = character.gender.toLowerCase();

                    if (genderCounts.hasOwnProperty(gender)) {

                        genderCounts[gender]++;

                    } else {

                        genderCounts.other++;

                    }

                }

                

                // Count objects

                if (character.object) {

                    objectCounts[character.object] = (objectCounts[character.object] || 0) + 1;

                }

                

                // Calculate average age and age groups

                if (character.age && !isNaN(character.age)) {

                    const age = parseInt(character.age);

                    totalAge += age;

                    ageCount++;

                    

                    if (age <= 17) ageGroups['0-17']++;

                    else if (age <= 20) ageGroups['18-20']++;

                    else if (age <= 25) ageGroups['21-25']++;

                    else if (age <= 30) ageGroups['26-30']++;

                    else if (age <= 35) ageGroups['31-35']++;

                    else if (age <= 40) ageGroups['36-40']++;

                    else if (age <= 45) ageGroups['41-45']++;

                    else if (age <= 50) ageGroups['46-50']++;

                    else ageGroups['51+']++;

                }

            });

            

            // Update gender stats

            document.getElementById('male-count').textContent = genderCounts.male;

            document.getElementById('female-count').textContent = genderCounts.female;

            document.getElementById('nb-count').textContent = genderCounts['non-binary'];

            

            // Calculate average age

            const averageAge = ageCount > 0 ? Math.round(totalAge / ageCount) : 0;

            document.getElementById('avg-age').textContent = averageAge;

            

            // Update charts

            updateGenderChart(genderCounts);

            updateAgeChart(ageGroups);

            updateObjectChart(objectCounts);

        }

        function handleImportedFile(event) {

            const file = event.target.files[0];

            if (!file) return;

            

            const reader = new FileReader();

            reader.onload = function(e) {

                try {

                    const importedData = JSON.parse(e.target.result);

                    if (Array.isArray(importedData)) {

                        if (confirm(`This will import ${importedData.length} characters. Continue?`)) {

                            characters = importedData;

                            localStorage.setItem('oc-characters', JSON.stringify(characters));

                            renderGallery();

                            updateObjectFilter();

                            updateStatistics();

                            alert('Data imported successfully!');

                        }

                    } else {

                        alert('Invalid data format. Expected an array of characters.');

                    }

                } catch (error) {

                    alert('Error parsing file: ' + error.message);

                }

            };

            reader.readAsText(file);

            

           

            event.target.value = '';

        }
